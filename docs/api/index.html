Quick Start

Get up and running with UMICP in minutes.

C++ Integration


#include "umicp/protocol.h"
#include "umicp/transport.h"

using namespace umicp;

// Create and configure protocol
Protocol protocol("my-node");
UMICPConfig config;
config.enable_binary = true;
config.require_encryption = true;
protocol.configure(config);

// Set up transport
auto transport = TransportFactory::create_websocket({
    .host = "localhost",
    .port = 8080
});
protocol.set_transport(transport);

// Connect and send data
protocol.connect();
ByteBuffer data = {0x01, 0x02, 0x03, 0x04};
protocol.send_data("target-node", data);

Node.js Integration


const { Protocol, TransportFactory } = require('umicp');

// Create protocol instance
const protocol = new Protocol('my-node');

// Configure with options
protocol.configure({
    enableBinary: true,
    requireEncryption: true,
    maxMessageSize: 1024 * 1024
});

// Set up WebSocket transport
const transport = TransportFactory.createWebSocket({
    host: 'localhost',
    port: 8080
});
protocol.setTransport(transport);

// Connect and send data
await protocol.connect();
const data = Buffer.from([0x01, 0x02, 0x03, 0x04]);
await protocol.sendData('target-node', data);

TypeScript Integration


import { Protocol, TransportFactory, UMICPConfig } from 'umicp';

// Create protocol with type safety
const protocol = new Protocol('my-node');

// Type-safe configuration
const config: UMICPConfig = {
    enableBinary: true,
    requireEncryption: true,
    maxMessageSize: 1024 * 1024,
    connectionTimeout: 5000
};
protocol.configure(config);

// Set up transport with type checking
const transport = TransportFactory.createWebSocket({
    host: 'localhost',
    port: 8080,
    path: '/umicp'
});
protocol.setTransport(transport);

// Connect and send with async/await
await protocol.connect();
const data = new Uint8Array([0x01, 0x02, 0x03, 0x04]);
await protocol.sendData('target-node', data);
